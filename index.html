import Parallaxy from "parallaxy-img";
import "parallaxy-img";

How to use ?

HTML ATTRIBUTES

                
[
    "data-prl-y",
    "data-prl-x",
    "data-prl-scale",
    "data-prl-breakpoint",
    "data-prl-adaptative",
    "data-prl-axes",
    "data-prl-speed-x",
    "data-prl-speed-y",
    "data-prl-overflow-x",
    "data-prl-overflow-y",
    "data-prl-inverted-x",
    "data-prl-inverted-y",
]

Without attributes in JS

                
const parallaxy = new Parallaxy({
    element: ...,
    'y': {
        speed: 0.5,
        invert: false,
        overflow: false
    },
    'x':{
        speed: 0.2,
        invert: false,
        overflow: false
    },
    adaptative: false, //Adapte the image to the container size (type -> query or number of parent to go up)
    axes: 200, //What is the axes of the parallax in px ? By default it's the middle screen
    breakPoint: "(max-width: 600px)", //Will stop parallax on mobile for example (default is false)
    scale: 1.5 //Need to be >= 1
});

parallaxy.addElement(...); //Add element to parallaxy configuration
parallaxy.removeElement(...); //Remove element from parallaxy configuration
parallaxy.start(); //Start parallaxy (actived by default)
parallaxy.stop(); //Stop parallaxy
parallaxy.reset(); //To remove transformation of each elements after stop

Important ! You need to center the image into the container

                
.container{
    display: flex;
    justify-content: center;
    align-items: center;
}
