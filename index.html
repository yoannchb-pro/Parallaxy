Parallaxy.js

Installation ?

NPM

                
npm i parallaxy-img

CDN

                
https://unpkg.com/parallaxy-img@2.0.0/dist/parallaxy.js

Import

                
const Parallaxy = require("parallaxy-img");
import Parallaxy from "parallaxy-img";
import "parallaxy-img";

Without attributes in JS

                
const parallaxy = new Parallaxy({
    element: ..., //one element only
    'y': {
        speed: 0.5,
        invert: false,
        overflow: false
    },
    'x':{
        speed: 0.2,
        invert: false,
        overflow: false
    },
    axes: 200, //What is the axes of the parallax in px ? By default it's the middle screen
    breakPoint: "(max-width: 600px)", //Will stop parallax on mobile for example (default is false)
    scale: 1.5 //Need to be >= 1
});

parallaxy.start(); //Start parallaxy (actived by default)
parallaxy.stop(); //Stop parallaxy
parallaxy.reset(); //To remove transformation of each elements after stop

Dont forget to center the image into the container

                
.container{
    display: flex;
    justify-content: center;
    align-items: center;
}
